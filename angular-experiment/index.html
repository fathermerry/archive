Experiment

Simple experiment to automate the writing of popular Angular JS ideas

The available controls are new module module-name, resolve dependency-name list/detail eg new module paygrades resolve paygrade list

angular.module('{{module.name}}', ['ui.router'])
.config(['$stateProvider', function($stateProvider) {
  $stateProvider
  .state('{{module.baseState}}', {
      url: '/{{module.url}}',
      templateUrl: 'components/{{module.name}}/{{module.name}}.html'
      controller: '{{module.ctrl}}'{{module.resolve ? ",
      resolve: {
        " + module.resolve.name + " : [" + module.resolve.paramss + "'Restangular', function(" + module.resolve.params + "Restangular){
          return Restangular." + module.resolve.count + "(" + module.resolve.url + ")." + module.resolve.method + "();
        }]
      }": ""}}
    });
}])
.controller('{{module.ctrl}}', [{{module.resolve ? module.resolve.name + ", " : ""}}'$scope', '$state', 'Restangular','notify', function({{module.resolve ? module.resolve.name + ", " : ""}}$scope, $state, Restangular, notify){
  {{module.resolve ? "$scope." + module.resolve.name + " = " + module.resolve.name : ""}}
}])
